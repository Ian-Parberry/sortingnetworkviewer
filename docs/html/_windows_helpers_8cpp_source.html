62void MinDragRect(HWND hwnd, WPARAM wParam, RECT* pRect, int n){
63 RECT cr; //client rectangle
64 RECT wr; //window rectangle, includes client rectangle and borders
65
66 GetClientRect(hwnd, &cr);
67 GetWindowRect(hwnd, &wr);
68
69 //combined border width and height
70
71 const int bw = (wr.right - wr.left) - (cr.right - cr.left); //border width
72 const int bh = (wr.bottom - wr.top) - (cr.bottom - cr.top); //border height
73
74 //new drag window width and height
75
76 const int dw = max(n, pRect->right - pRect->left - bw) + bw; //new width
77 const int dh = max(n, pRect->bottom - pRect->top - bh) + bh; //new height
78
79 //enforce new drag window width and height
80
81 switch(wParam){ //which edge are we dragging on?
82 case WMSZ_LEFT: //left edge
83 pRect->left = pRect->right - dw;
84 break;
85
86 case WMSZ_RIGHT: //right edge
87 pRect->right = pRect->left + dw;
88 break;
89
90 case WMSZ_TOP: //top edge
91 pRect->top = pRect->bottom - dh;
92 break;
93
94 case WMSZ_BOTTOM: //bottom edge
95 pRect->bottom = pRect->top + dh;
96 break;
97
98 case WMSZ_TOPRIGHT: //top right corner
99 pRect->top = pRect->bottom - dh;
100 pRect->right = pRect->left + dw;
101 break;
102
103 case WMSZ_TOPLEFT: //top left corner
104 pRect->top = pRect->bottom - dh;
105 pRect->left = pRect->right - dw;
106 break;
107
108 case WMSZ_BOTTOMRIGHT: //bottom right corner
109 pRect->bottom = pRect->top + dh;
110 pRect->right = pRect->left + dw;
111 break;
112
113 case WMSZ_BOTTOMLEFT: //bottom left corner
114 pRect->bottom = pRect->top + dh;
115 pRect->left = pRect->right - dw;
116 break;
117 } //switch
118} //MinDragRect

191HRESULT GetEncoderClsid(const WCHAR* format, CLSID* pClsid){
192 UINT num = 0; //number of image encoders
193 UINT n = 0; //size of the image encoder array in bytes
194 HRESULT hr = E_FAIL; //return result
195
196 Gdiplus::ImageCodecInfo* pCodecInfo = nullptr; //for codec info
197 if(FAILED(Gdiplus::GetImageEncodersSize(&num, &n)))return E_FAIL; //get sizes
198 if(n == 0)return E_FAIL; //there are no encoders
199
200 pCodecInfo = (Gdiplus::ImageCodecInfo*)(malloc(n)); //allocate codec info memory
201 if(pCodecInfo == nullptr)return E_FAIL; //malloc failed (as if)
202 if(FAILED(GetImageEncoders(num, n, pCodecInfo)))return E_FAIL; //get encoders
203
204 for(UINT j=0; j<num && hr!=S_OK; j++) //for each encoder, while not found
205 if(wcscmp(pCodecInfo[j].MimeType, format) == 0){ //found the codex we want
206 *pClsid = pCodecInfo[j].Clsid; //return it
207 hr = S_OK; //success
208 } //if
209
210 free(pCodecInfo); //clean up
211 return hr;
212} //GetEncoderClsid

223HRESULT ExportImage(const eExport t, HWND hwnd, CRenderableComparatorNet* pNet,
224 std::wstring& wstrName)
225{
226 COMDLG_FILTERSPEC filetypes[1];
227 std::wstring wstrTitle = L"Export "; //for dialog box title
228 std::wstring wstrDefaultExtension; //for dialog box default file extension
229
230 //set file types, title, and default extension for export type
231
232 switch(t){
233 case eExport::Png:
234 filetypes[0] = {L"PNG Files", L"*.png"};
235 wstrTitle += L"PNG"; wstrDefaultExtension = L"png";
236 break;
237
238 case eExport::Svg:
239 filetypes[0] = {L"SVG Files", L"*.svg"};
240 wstrTitle += L"SVG"; wstrDefaultExtension = L"svg";
241 break;
242
243 case eExport::TeX:
244 filetypes[0] = {L"TeX Files", L"*.tex"};
245 wstrTitle += L"TeX"; wstrDefaultExtension = L"tex";
246 break;
247 } //switch
248
249 wstrTitle += L" Image";
250
251 //prepare for the dialog box
252
253 CComPtr<IFileSaveDialog> pDlg; //pointer to save dialog box
254 CComPtr<IShellItem> pItem; //item pointer
255 LPWSTR pwsz = nullptr; //pointer to null-terminated wide string for result
256
257 //fire up the save dialog box
258
259 HRESULT hr = pDlg.CoCreateInstance(__uuidof(FileSaveDialog));
260
261 if(SUCCEEDED(hr)){
262 pDlg->SetFileTypes(_countof(filetypes), filetypes); //set file types
263 pDlg->SetTitle(wstrTitle.c_str()); //set title bar text
264 pDlg->SetFileName(wstrName.c_str()); //set default file name
265 pDlg->SetDefaultExtension(wstrDefaultExtension.c_str()); //set default extension
266
267 hr = pDlg->Show(hwnd); //show the dialog box
268
269 if(SUCCEEDED(hr)){
270 hr = pDlg->GetResult(&pItem); //get the result item
271
272 if(SUCCEEDED(hr)){
273 hr = pItem->GetDisplayName(SIGDN_FILESYSPATH, &pwsz); //get file name
274
275 if(SUCCEEDED(hr)){
276 switch(t){ //here's where the export happens
277 case eExport::Png: hr = pNet->ExportToPNG(pwsz); break;
278 case eExport::Svg: hr = pNet->ExportToSVG(pwsz); break;
279 case eExport::TeX: hr = pNet->ExportToTex(pwsz); break;
280 } //switch
281
282 if(SUCCEEDED(hr))
283 wstrName = FileNameBase(std::wstring(pwsz)); //set file name
284 } //if
285 } //if
286 } //if
287 } //if
288
289 CoTaskMemFree(pwsz); //clean up
290
291 return hr;
292} //ExportImage
