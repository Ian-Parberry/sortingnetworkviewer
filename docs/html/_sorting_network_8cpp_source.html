68UINT CSortingNetwork::flipinput(UINT j, const UINT firstlayer, const UINT lastlayer){
69 for(UINT i=firstlayer; i<=lastlayer; i++){ //for each layer in range
70 m_nValue[i][j] ^= 1; //flip the value on channel j at that level
71
72 UINT k = m_nMatch[i][j]; //find the channel to which it is joined via a comparator, if any
73
74 if(0 <= k && k < m_nInputs)
75 if((m_nValue[i][k] && j>k) || !(m_nValue[i][k] || j>k))
76 j = k;
77 } //for
78
79 return j;
80} //flipinput

94bool CSortingNetwork::sorts(){
95 UINT i=0; //index of bit to flip
96 bool bSorts = true; //assume it sorts until we find otherwise
97 initSortingTest(); //intialize input and values in comparator network to zero
98
99 while(bSorts && i<=m_nInputs){ //bail if it doesn't sort, or we've tried all binary inputs
100 i = m_pGrayCode->Next(); //next bit to flip in Gray code order
101 bSorts = bSorts && (i>m_nInputs || stillsorts(i)); //check whether it still sorts when this bit is flipped
102 } //while
103
104 return bSorts;
105} //sorts
