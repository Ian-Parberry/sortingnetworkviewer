42Gdiplus::REAL CRenderableComparatorNet::ComputeBitmapHeight(){
43 bool* bUsed = new bool[m_nInputs]; //whether channels are used at the current level
44 Gdiplus::REAL fHeight = m_fYDelta + m_fYDelta2; //height of comparator network so far
45
46 for(UINT i=0; i<m_nDepth; i++){
47 for(UINT j=0; j<m_nInputs; j++) //mark all channels unused
48 bUsed[j] = false;
49
50 //make a pass through the current level printing non-overlapping comparators
51
52 for(UINT pass=0; pass<m_nInputs/2; pass++){
53 bool bPassUsed = false; //whether unprinted comparator found in this level
54
55 for(UINT j=0; j<m_nInputs; j++){ //for rach channel
56 const UINT dest = m_nMatch[i][j]; //other end of comparator
57
58 if(dest < m_nInputs && dest > j && !bUsed[j]){ //can print without overlap
59 bPassUsed = bUsed[j] = bUsed[dest] = true; //mark that we've printed it
60 j = dest; //so that later comparators can't overlap
61 } //if
62 } //for
63
64 if(bPassUsed)fHeight += m_fYDelta; //current pass found a comparator
65 } //for
66
67 fHeight += m_fYDelta2; //next level
68 } //for
69
70 delete [] bUsed;
71 return fHeight;
72} //ComputeBitmapHeight

84void CRenderableComparatorNet::DrawComparator(
85 const UINT src, const UINT dest, const float fDist)
86{
87 float fSrcy = 0, fDesty = 0, fSrcx = 0, fDestx = 0; //end points for PNG, SVG
88 int vx = 0, vy = 0; //axis for TeX
89 const float r = m_fDiameter/2.0f; //circle radius for connectors
90
91 switch(m_eDrawStyle){
92 case eDrawStyle::Vertical:
93 fSrcx = m_fDiameter/2 + src*m_fXDelta;
94 fDestx = m_fDiameter/2 + dest*m_fXDelta;
95 fSrcy = fDesty = fDist;
96 vx = 1; vy = 0;
97 break;
98
99 case eDrawStyle::Horizontal:
100 fSrcx = fDestx = fDist;
101 fSrcy = m_fDiameter/2 + src*m_fXDelta;
102 fDesty = m_fDiameter/2 + dest*m_fXDelta;
103 vx = 0; vy = -1;
104 break;
105 } //switch
106
107 const int nSrcx = (UINT)std::round(fSrcx);
108 const int nSrcy = (UINT)std::round(fSrcy);
109
110 const int nDestx = (UINT)std::round(fDestx);
111 const int nDesty = (UINT)std::round(fDesty);
112
113 switch(m_eExportType){
114 case eExport::Png:
115 if(m_pGraphics && m_pPen && m_pBrush){
116 const float d = m_fDiameter; //shorthand for diameter
117
118 m_pGraphics->FillEllipse(m_pBrush, fSrcx - r, fSrcy - r, d, d);
119 m_pGraphics->FillEllipse(m_pBrush, fDestx - r, fDesty - r, d, d);
120 m_pGraphics->DrawLine(m_pPen, fSrcx, fSrcy, fDestx, fDesty);
121 } //if
122 break;
123
124 case eExport::Svg:
125 if(m_pOutput){
126 fprintf_s(m_pOutput, "<circle cx=\"%d\" cy=\"%d\"/>", nSrcx, nSrcy);
127 fprintf_s(m_pOutput, "<circle cx=\"%d\" cy=\"%d\"/>", nDestx, nDesty);
128 fprintf_s(m_pOutput, "<line x1=\"%d\" y1=\"%d\" x2=\"%d\" y2=\"%d\"/>\n",
129 nSrcx, nSrcy, nDestx, nDesty);
130 } //if
131 break;
132
133 case eExport::TeX:
134 if(m_pOutput){
135 const int d = (UINT)std::round(m_fDiameter);
136 const UINT nLen = abs(nDestx - nSrcx + nDesty - nSrcy);
137
138 fprintf_s(m_pOutput, "\\put(%d,-%d){\\circle*{%d}}\n", nSrcx, nSrcy, d);
139 fprintf_s(m_pOutput, "\\put(%d,-%d){\\circle*{%d}}\n", nDestx, nDesty, d);
140 fprintf_s(m_pOutput, "\\put(%d,-%d){\\line(%d,%d){%u}}\n", nDestx, nDesty,
141 vx, vy, nLen);
142 } //if
143 break;
144 } //switch
145} //DrawComparator

155void CRenderableComparatorNet::DrawComparators(){
156 bool* bUsed = new bool[m_nInputs]; //whether channels are used at the current level
157 float fLen = m_fYDelta + m_fYDelta2; //distance along channel
158
159 for(UINT i=0; i<m_nDepth; i++){
160 for(UINT j=0; j<m_nInputs; j++) //mark all channels unused
161 bUsed[j] = false;
162
163 //make a pass through the current level printing comparators that don't overlap
164
165 for(UINT pass=0; pass<m_nInputs/2; pass++){
166 bool bPassUsed = false; //whether we found an unprinted comparator in this level
167
168 switch(m_eDrawStyle){
169 case eDrawStyle::Vertical:
170 for(UINT j=0; j<m_nInputs; j++){
171 const UINT dest = m_nMatch[i][j];
172
173 if(dest < m_nInputs && dest > j && !bUsed[j]){ //can print without overlap
174 DrawComparator(dest, j, fLen); //draw comparator
175 bPassUsed = bUsed[j] = bUsed[dest] = true; //mark that we've printed it
176 j = dest; //so that later comparators can't overlap
177 } //if
178 } //for
179 break;
180
181 case eDrawStyle::Horizontal:
182 for(int j=m_nInputs-1; j>=0; j--){
183 const UINT dest = m_nMatch[i][j];
184
185 if(dest < m_nInputs && dest < (UINT)j && !bUsed[dest]){ //can print without overlap
186 DrawComparator(j, dest, fLen); //draw comparator
187 bPassUsed = bUsed[j] = bUsed[dest] = true; //mark that we've printed it
188 j = dest; //so that later comparators can't overlap
189 } //if
190 } //for
191 break;
192 } //switch
193
194 if(bPassUsed)fLen += m_fYDelta; //the current pass found a comparator
195 } //for
196
197 fLen += m_fYDelta2; //next level
198 } //for
199
200 delete [] bUsed;
201} //DrawComparators

211void CRenderableComparatorNet::DrawChannels(const float fLen){
212 float fSrcy = 0, fDesty = 0, fSrcx = 0, fDestx = 0; //end points for PNG, SVG
213 int vx = 0, vy = 0; //axis for TeX
214
215 switch(m_eDrawStyle){
216 case eDrawStyle::Vertical:
217 fSrcy = 0.0f; fDesty = fLen;
218 fSrcx = m_fDiameter/2; fDestx = fSrcx;
219 vx = 0; vy = -1;
220 break;
221
222 case eDrawStyle::Horizontal:
223 fSrcx = 0.0f; fDestx = fLen;
224 fSrcy = m_fDiameter/2; fDesty = fSrcy;
225 vx = 1; vy = 0;
226 break;
227 } //switch
228
229 const int nLen = (UINT)std::round(fLen);
230
231 for(UINT i=0; i<m_nInputs; i++){
232 const int nSrcx = (UINT)std::round(fSrcx);
233 const int nSrcy = (UINT)std::round(fSrcy);
234
235 const int nDestx = (UINT)std::round(fDestx);
236 const int nDesty = (UINT)std::round(fDesty);
237
238 switch(m_eExportType){
239 case eExport::Png:
240 if(m_pGraphics) //safety
241 m_pGraphics->DrawLine(m_pPen, fSrcx, fSrcy, fDestx, fDesty);
242 break;
243
244 case eExport::Svg:
245 if(m_pOutput) //safety
246 fprintf_s(m_pOutput, "<line x1=\"%d\" y1=\"%d\" x2=\"%d\" y2=\"%d\"/>\n",
247 nSrcx, nSrcy, nDestx, nDesty);
248 break;
249
250 case eExport::TeX:
251 if(m_pOutput) //safety
252 fprintf_s(m_pOutput, "\\put(%d,-%d){\\line(%d,%d){%d}}\n",
253 nSrcx, nSrcy, vx, vy, nLen);
254 break;
255 } //switch
256
257 switch(m_eDrawStyle){
258 case eDrawStyle::Vertical: fSrcx += m_fXDelta; fDestx = fSrcx; break;
259 case eDrawStyle::Horizontal: fSrcy += m_fXDelta; fDesty = fSrcy; break;
260 } //switch
261 } //for
262} //DrawChannels

274void CRenderableComparatorNet::Draw(const eDrawStyle d){
275 m_eExportType = eExport::Png;
276 m_eDrawStyle = d;
277
278 const UINT w = (UINT)std::ceil((m_nInputs - 1)*m_fXDelta + m_fPenWidth + m_fDiameter);
279 const UINT h = (UINT)std::ceil(ComputeBitmapHeight());
280
281 switch(m_eDrawStyle){
282 case eDrawStyle::Vertical: m_pBitmap = new Gdiplus::Bitmap(w, h); break;
283 case eDrawStyle::Horizontal: m_pBitmap = new Gdiplus::Bitmap(h, w); break;
284 } //switch
285
286 m_pGraphics = new Gdiplus::Graphics(m_pBitmap);
287 m_pGraphics->SetSmoothingMode(Gdiplus::SmoothingModeHighQuality);
288 m_pGraphics->Clear(Gdiplus::Color::Transparent);
289 m_pPen = new Gdiplus::Pen(Gdiplus::Color::Black);
290 m_pPen->SetWidth(m_fPenWidth);
291 m_pBrush = new Gdiplus::SolidBrush(Gdiplus::Color::Black);
292
293 DrawChannels((float)h);
294 DrawComparators();
295
296 delete m_pGraphics; m_pGraphics = nullptr;
297 delete m_pPen; m_pPen = nullptr;
298 delete m_pBrush; m_pBrush = nullptr;
299} //Draw
