73void CMain::OnPaint(){
74 PAINTSTRUCT ps; //paint structure
75 HDC hdc = BeginPaint(m_hWnd, &ps); //device context
76 Gdiplus::Graphics graphics(hdc); //GDI+ graphics object
77 graphics.Clear(Gdiplus::Color::White); //clear to white
78
79 Gdiplus::Bitmap* pBitmap = GetBitmap();
80
81 if(pBitmap){
82 const int w = pBitmap->GetWidth(); //bitmap width
83 const int h = pBitmap->GetHeight(); //bitmap height
84
85 //get client area width and height
86
87 RECT rectClient; //for client rectangle
88 GetClientRect(m_hWnd, &rectClient); //get client rectangle
89 const int nClientWidth = rectClient.right - rectClient.left; //client width
90 const int nClientHeight = rectClient.bottom - rectClient.top; //client height
91
92 //compute the drawing scale for the bitmap
93
94 const UINT margin = 20; //margin width
95
96 const float xscale = float(nClientWidth - 2*margin)/w; //horizontal scale
97 const float yscale = float(nClientHeight - 2*margin)/h; //vertical scale
98 const float scale = min(min(xscale, yscale), 1); //actual scale
99
100 //compute the destination rectangle
101
102 Gdiplus::Rect rectDest; //destination rectangle
103
104 rectDest.Width = (int)std::floor(scale*w);
105 rectDest.Height = (int)std::floor(scale*h);
106
107 rectDest.X = max(margin, (nClientWidth - rectDest.Width)/2);
108 rectDest.Y = max(margin, (nClientHeight - rectDest.Height)/2);
109
110 //draw the bitmap to the window
111
112 graphics.DrawImage(pBitmap, rectDest);
113 } //if
114
115 EndPaint(m_hWnd, &ps); //this must be done last
116} //OnPaint

192template<class t> void CMain::Generate(){
193 UINT n = 0; //for number of inputs
194
195 if(SUCCEEDED(CDialogBox().GetNumInputs(m_hWnd, n))){ //got number of inputs
196 delete m_pSortingNetwork; //delete old sorting network
197
198 t* p = new t(CeilLog2(n)); //sorting network with inputs rounded up to next power of 2
199 if(!IsPowerOf2(n))p->Prune(n); //prune unneeded channels and comparators
200 m_wstrName = p->GetName(); //get name
201 m_pSortingNetwork = p; //this is the new sorting network
202
203 EnableMenus();
204 } //if
205} //Generate

241void CMain::Verify(){
242 if(m_pSortingNetwork == nullptr)return; //bail out
243
244 const std::string strInputs = std::to_string(m_pSortingNetwork->GetNumInputs());
245 const std::string strDepth = std::to_string(m_pSortingNetwork->GetDepth());
246 const std::string strSize = std::to_string(m_pSortingNetwork->GetSize());
247
248 const std::string strDetails = "of size " + strSize + " and depth " + strDepth;
249
250 UINT nIcon = 0; //for icon flag in message box
251 std::string s; //for message box text
252
253 bool bSorts = false; //true if it sorts
254 bool bUnknown = false; //true if sorting status is unknown
255
256 if(m_pSortingNetwork->GetNumInputs() < 30)
257 bSorts = m_pSortingNetwork && m_pSortingNetwork->sorts();
258
259 else{
260 s = "Sorting network verification is Co-NP-complete.";
261 s += " This may take a long time. Are you sure you want to proceed?";
262
263 const int id = MessageBox(nullptr, s.c_str(), "Verify", MB_ICONQUESTION | MB_YESNO);
264
265 if(id == IDYES)
266 bSorts = m_pSortingNetwork && m_pSortingNetwork->sorts();
267 else bUnknown = true;
268 } //else
269
270 if(bSorts){
271 s = "This is a " + strInputs + "-input sorting network " + strDetails + ".";
272 nIcon = MB_ICONINFORMATION;
273 } //if
274
275 else if(bUnknown){
276 s = "This is a " + strInputs + "-input comparator network " + strDetails + ".";
277 nIcon = MB_ICONINFORMATION;
278 } //else if
279
280 else{
281 s = "This is a " + strInputs + "-input comparator network " + strDetails +
282 " that is not a sorting network.";
283 nIcon = MB_ICONERROR;
284 } //else
285
286 if(m_pSortingNetwork && m_pSortingNetwork->FirstNormalForm())
287 s += " It is in First Normal Form.";
288 else s += " It is not in First Normal Form.";
289
290 MessageBox(nullptr, s.c_str(), "Verify", nIcon | MB_OK);
291} //Verify
